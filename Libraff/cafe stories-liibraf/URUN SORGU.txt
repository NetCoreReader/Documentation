			
						
							
							SELECT  RM.KODU, RKSS.ACIKLAMA AS ‹STKATEGOR›,RK.ACIKLAMA ALTKATEGOR›,
							  RM.ACIKLAMASI,RM.YAYINCISI,RM.SINIFI  
 							     FROM RBS_051_MALZEME RM 
								LEFT OUTER JOIN RBS_051_KATEGORI RK ON RM.KATEGORI_ID=RK.ID  
									LEFT OUTER JOIN RBS_051_MALZEME_BIRIMI RMB ON RM.ANABIRIM_ID=RMB.ANABIRIM 
										LEFT OUTER JOIN RBS_051_BARKOD RKK ON RMB.ITEM_ID=RKK.ITEM_ID 
											LEFT OUTER JOIN RBS_051_KATEGORI RKSS ON RK.UST_ID=RKSS.ID
														WHERE RM.WEB_AKTARILDI=1 AND RM.DURUMU=0
